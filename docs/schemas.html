Schemas JSON

Definición completa de todos los objetos que devuelve y acepta la API.

User

json schema 
{
  "id":       "integer — read only",
  "username": "string — único, requerido",
  "email":    "string (email) — requerido",
  "phone":    "string — opcional, máx. 20 chars",
  "role":     "enum: CLIENT | ADMIN"
}

Product

json schema 
{
  "id":           "integer — read only",
  "name":         "string — requerido, máx. 100 chars",
  "product_type": "enum: TRADICIONAL | QUESO | CHOCLO | RELLENA | MORADA",
  "description":  "string — requerido",
  "price":        "decimal string — ej: '4500.00', máx. 10 dígitos, 2 decimales",
  "stock":        "integer — ≥ 0",
  "is_active":    "boolean — default: true",
  "image":        "string (ruta relativa) | null — write only en POST",
  "image_url":    "string (URL absoluta) | null — read only",
  "created_at":   "datetime ISO 8601 — read only",
  "updated_at":   "datetime ISO 8601 — read only"
}

Order (lectura)

json schema 
{
  "id":               "integer — read only",
  "payment_method":   "enum: MANUAL | GATEWAY",
  "delivery_address": "string — requerido, no puede estar vacío",
  "total":            "decimal string — calculado por el servidor, read only",
  "status":           "enum: PENDING | PAID | SHIPPED | CANCELLED — read only en creación",
  "created_at":       "datetime ISO 8601 — read only",
  "order_items":      "array<OrderItem> — read only"
}

Order (creación — request body)

json schema 
{
  "payment_method":   "enum: MANUAL | GATEWAY — requerido",
  "delivery_address": "string — requerido, no puede ser vacío ni solo espacios",
  "items": [
    {
      "product_id": "integer — requerido",
      "quantity":   "integer — requerido, ≥ 1, ≤ stock disponible"
    }
  ]
}

OrderItem

json schema 
{
  "id":           "integer — read only",
  "product":      "integer (product ID)",
  "product_name": "string — read only",
  "quantity":     "integer — ≥ 1",
  "unit_price":   "decimal string — precio al momento de la compra, read only"
}

PaymentProof

json schema 
{
  "id":          "integer — read only",
  "order":       "integer (order ID) — read only",
  "file":        "string (URL) — archivo subido",
  "status":      "enum: PENDING | APPROVED | REJECTED",
  "uploaded_at": "datetime ISO 8601 — read only"
}
